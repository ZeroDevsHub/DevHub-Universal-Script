local function updateCrosshair()
    -- Return immediately when disabled — prevents scroll blocking and position flicker
    if not crosshairEnabled then
        for i=1,4 do chLines[i].Visible=false end
        chDot.Visible=false; return
    end
    -- Skip while a TextBox is focused (e.g. Rayfield inputs)
    if UserInputService:GetFocusedTextBox() then return end
    local cx=Camera.ViewportSize.X/2; local cy=Camera.ViewportSize.Y/2
    local g=crosshairGap; local s=crosshairSize
    chLines[1].From=Vector2.new(cx,cy-g); chLines[1].To=Vector2.new(cx,cy-g-s)
    chLines[2].From=Vector2.new(cx,cy+g); chLines[2].To=Vector2.new(cx,cy+g+s)
    chLines[3].From=Vector2.new(cx-g,cy); chLines[3].To=Vector2.new(cx-g-s,cy)
    chLines[4].From=Vector2.new(cx+g,cy); chLines[4].To=Vector2.new(cx+g+s,cy)
    for i=1,4 do chLines[i].Color=crosshairColor; chLines[i].Visible=true end
    chDot.Position=Vector2.new(cx,cy); chDot.Color=crosshairColor; chDot.Visible=true
end

local function buildHighlight(player)
    if player==LocalPlayer or not player.Character then return end
    if isTeammate(player) or not espEnabled then return end
    local d=getOrCreate(player)
    if d.highlight then pcall(function() d.highlight:Destroy() end); d.highlight=nil end
    local hl=Instance.new("Highlight"); hl.Adornee=player.Character
    hl.FillColor=espEnemyCol; hl.OutlineColor=espEnemyCol
    hl.FillTransparency=0.60; hl.OutlineTransparency=0.05
    hl.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop; hl.Parent=player.Character; d.highlight=hl
end
local function rebuildAllHighlights()
    for _,p in pairs(Players:GetPlayers()) do
        if p~=LocalPlayer then
            local d=drawings[p.Name]
            if d and d.highlight then pcall(function() d.highlight:Destroy() end); d.highlight=nil end
            buildHighlight(p)
        end
    end
end

local function updatePlayerESP(player)
    if player==LocalPlayer then return end
    local d=getOrCreate(player)
    local alive=isAlive(player); local mate=isTeammate(player)
    local show=alive and not (mate and espTeamCheck)
    local hrp=getRootPart(player)
    local col=mate and espFriendCol or espEnemyCol

    local sb=nil
    if show and hrp then
        local headP=getHead(player)
        local topPos=headP and (headP.Position+Vector3.new(0,0.55,0)) or (hrp.Position+Vector3.new(0,3,0))
        local botPos=hrp.Position-Vector3.new(0,3.05,0)
        local top,tv=Camera:WorldToViewportPoint(topPos)
        local bot,bv=Camera:WorldToViewportPoint(botPos)
        local dist=(hrp.Position-Camera.CFrame.Position).Magnitude
        if tv and bv and top.Z>0 and bot.Z>0 and dist<=espMaxDist then
            local bH=math.abs(top.Y-bot.Y); local bW=bH*0.58
            sb={x=top.X-bW/2, y=math.min(top.Y,bot.Y), w=bW, h=bH, dist=dist}
        end
    end

    if espBoxEnabled and sb then
        d.box.Size=Vector2.new(sb.w,sb.h); d.box.Position=Vector2.new(sb.x,sb.y)
        d.box.Color=col; d.box.Thickness=espBoxThick; d.box.Visible=true
    else d.box.Visible=false end

    local BAR_W,BAR_GAP=4,3
    if espHealthEnabled and sb then
        local hum=getHumanoid(player)
        local pct=hum and math.clamp(hum.Health/math.max(hum.MaxHealth,1),0,1) or 1
        local fullH=sb.h; local fillH=math.max(1,fullH*pct)
        local barX=sb.x-BAR_W-BAR_GAP
        d.hpBg.Size=Vector2.new(BAR_W,fullH); d.hpBg.Position=Vector2.new(barX,sb.y); d.hpBg.Visible=true
        d.hpBar.Size=Vector2.new(BAR_W,fillH); d.hpBar.Position=Vector2.new(barX,sb.y+fullH-fillH)
        d.hpBar.Color=Color3.fromRGB(0,215,75); d.hpBar.Visible=true
    else d.hpBg.Visible=false; d.hpBar.Visible=false end

    if espDistEnabled and sb then
        d.distText.Text=math.floor(sb.dist).."m"
        d.distText.Position=Vector2.new(sb.x+sb.w/2, sb.y-16)
        d.distText.Color=Color3.fromRGB(195,215,255); d.distText.Visible=true
    else d.distText.Visible=false end

    if espNameEnabled and sb then
        d.nameText.Text=player.Name
        d.nameText.Position=Vector2.new(sb.x+sb.w/2, sb.y-(espDistEnabled and -29 or -16))
        d.nameText.Color=col; d.nameText.Visible=true
    else d.nameText.Visible=false end

    if espTracerEnabled and show and hrp then
        local footPos=hrp.Position-Vector3.new(0,3.05,0)
        local sp,on=Camera:WorldToViewportPoint(footPos)
        local dist=(hrp.Position-Camera.CFrame.Position).Magnitude
        if on and sp.Z>0 and dist<=espMaxDist then
            local vp=Camera.ViewportSize
            d.tracer.From=Vector2.new(vp.X/2,vp.Y); d.tracer.To=Vector2.new(sp.X,sp.Y)
            d.tracer.Color=col; d.tracer.Thickness=1; d.tracer.Visible=true
        else d.tracer.Visible=false end
    else d.tracer.Visible=false end

    local used=0
    if espSkeletonEnabled and show and player.Character then
        local char=player.Character
        for _,pair in ipairs(SKEL) do
            local pA=char:FindFirstChild(pair[1]); local pB=char:FindFirstChild(pair[2])
            if pA and pB then
                local spA,onA=Camera:WorldToViewportPoint(pA.Position)
                local spB,onB=Camera:WorldToViewportPoint(pB.Position)
                if onA and onB and spA.Z>0 and spB.Z>0 then
                    used=used+1; local sl=d.skel[used]
                    if sl then sl.From=Vector2.new(spA.X,spA.Y); sl.To=Vector2.new(spB.X,spB.Y); sl.Color=espSkelCol; sl.Thickness=1; sl.Visible=true end
                end
            end
        end
    end
    for i=used+1,MAX_SKEL do local sl=d.skel[i]; if sl then sl.Visible=false end end

    if d.highlight then
        if d.highlight.Parent==nil or d.highlight.Adornee~=player.Character then
            pcall(function() d.highlight:Destroy() end); d.highlight=nil
            if espEnabled and not mate and player.Character then
                local hl=Instance.new("Highlight"); hl.Adornee=player.Character
                hl.FillColor=col; hl.OutlineColor=col; hl.FillTransparency=0.60; hl.OutlineTransparency=0.05
                hl.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop; hl.Parent=player.Character; d.highlight=hl
            end
        end
    end
end

-- ══════════════════════════════════════
-- AIMBOT TARGET FINDER
-- Returns the closest enemy BasePart within aimbotFOV pixels of screen center.
-- "fovRadius" param lets us reuse this for silent aim with a tighter radius.
-- ══════════════════════════════════════
local function getBestTarget(fovRadius)
    fovRadius = fovRadius or aimbotFOV
    local best, bestDist, bestPos = nil, fovRadius, nil
    local vp  = Camera.ViewportSize
    local ctr = Vector2.new(vp.X/2, vp.Y/2)

    for _,p in pairs(Players:GetPlayers()) do
        if p==LocalPlayer or isTeammate(p) or not isAlive(p) then continue end
        local bone=getBone(p); if not bone then continue end

        local worldPos = bone.Position
        -- Optional velocity prediction for aimbot
        if aimbotPrediction and getRootPart(p) then
            worldPos = worldPos + getRootPart(p).Velocity * predictionAmount
        end

        local sp, onScreen = Camera:WorldToViewportPoint(worldPos)
        if not onScreen or sp.Z <= 0 then continue end

        local screenPos = Vector2.new(sp.X, sp.Y)
        local dist      = (screenPos - ctr).Magnitude
        if dist < bestDist then
            bestDist = dist
            best     = bone
            bestPos  = worldPos
        end
    end
    return best, bestPos
end

-- ══════════════════════════════════════
-- NO RECOIL — full CFrame delta approach
-- Records camera CFrame at the END of each frame.
-- Next frame, if camera was kicked by more than threshold in any direction,
-- we restore only the recoil delta — leaving intentional mouse movement intact.
-- ══════════════════════════════════════
local function applyNoRecoil()
    if not noRecoilEnabled then noRecoilBaseCF=nil; return end

    local currentCF = Camera.CFrame

    if noRecoilBaseCF == nil then
        -- First frame after enabling — just store baseline
        noRecoilBaseCF = currentCF
        return
    end

    -- Decompose both CFrames into euler angles
    local prevP, prevY, prevR = noRecoilBaseCF:ToEulerAnglesYXZ()
    local currP, currY, currR = currentCF:ToEulerAnglesYXZ()

    -- Delta in pitch and yaw this frame
    local dPitch = currP - prevP
    local dYaw   = currY - prevY

    local corrected = false
    local newP, newY = currP, currY

    -- Recoil kicks pitch upward = positive pitch increase beyond threshold
    if dPitch > noRecoilThreshold then
        newP = prevP  -- cancel the recoil pitch kick
        corrected = true
    end
    -- Recoil can also add a yaw component (spray pattern side-to-side)
    -- We only cancel yaw if the kick is significant AND player isn't actively
    -- moving their mouse horizontally (detected by checking if dYaw is sudden)
    if math.abs(dYaw) > noRecoilThreshold * 0.5 and math.abs(dYaw) < 0.15 then
        -- Small sudden yaw = recoil spray. Large yaw = player turning (don't cancel)
        newY = prevY
        corrected = true
    end

    if corrected then
        Camera.CFrame = CFrame.new(currentCF.Position)
            * CFrame.fromEulerAnglesYXZ(newP, newY, prevR)
    end

    -- Update baseline to corrected CFrame for next frame
    noRecoilBaseCF = Camera.CFrame
end

-- ══════════════════════════════════════
-- SILENT AIM — proximity-check + one-frame snap
-- Only fires when the crosshair is ALREADY near an enemy (within silentAimFOV).
-- The camera snaps to target for one frame (so the bullet ray hits),
-- then restores immediately. Your crosshair visually stays where you aimed.
-- This means: aim near someone's head → shoot → bullet hits head → they die.
-- ══════════════════════════════════════
local silentSnapping = false

UserInputService.InputBegan:Connect(function(inp, gp)
    -- Only fire on LMB click, silent aim enabled, not blocked by game input
    if inp.UserInputType ~= Enum.UserInputType.MouseButton1 then return end
    if not silentAimEnabled then return end
    if gp then return end
    if silentSnapping then return end  -- don't double-fire

    -- Find an enemy within the TIGHT silent aim radius (not the full aimbot FOV)
    local bone, worldPos = getBestTarget(silentAimFOV)
    if not bone or not worldPos then return end  -- crosshair not near anyone — do nothing

    silentSnapping = true
    local savedCF = Camera.CFrame
    local origin  = savedCF.Position

    -- Snap camera so bullet ray points at enemy head
    Camera.CFrame = CFrame.lookAt(origin, worldPos)

    -- Restore BEFORE the next render frame so the visual camera never visibly moves.
    -- task.wait(0) yields to the next heartbeat which is before RenderStepped,
    -- ensuring the game's raycast sees our snapped angle but the player never sees it.
    task.defer(function()
        Camera.CFrame = savedCF
        silentSnapping = false
    end)
end)

end

-- TRIGGERBOT
-- ══════════════════════════════════════
local lastTrigger=0
local function doTriggerbot()
    if not triggerbotEnabled then return end
    local now=tick(); if now-lastTrigger<triggerbotDelay then return end
    local char=LocalPlayer.Character; if not char then return end
    local cf=Camera.CFrame; local ray=Ray.new(cf.Position,cf.LookVector*600)
    local hit=workspace:FindPartOnRayWithIgnoreList(ray,{char}); if not hit then return end
    local model=hit:FindFirstAncestorOfClass("Model"); if not model then return end
    for _,p in pairs(Players:GetPlayers()) do
        if p~=LocalPlayer and p.Character==model and not isTeammate(p) and isAlive(p) then
            lastTrigger=now
            pcall(function()
                local vi=game:GetService("VirtualInputManager")
                vi:SendMouseButtonEvent(0,0,0,true,game,0); task.wait(0.05)
                vi:SendMouseButtonEvent(0,0,0,false,game,0)
            end)
            break
        end
    end
end

-- ══════════════════════════════════════
-- ANTI-FLASHBANG (event-driven, zero per-frame cost)
-- ══════════════════════════════════════
local flashConns={}
local function killFlashObj(obj)
    if obj:IsA("Frame") or obj:IsA("ImageLabel") then
        if obj.BackgroundTransparency<0.1 then
            local c=obj.BackgroundColor3
            if c.R>0.85 and c.G>0.85 and c.B>0.85 then obj.BackgroundTransparency=1 end
        end
    elseif obj:IsA("ColorCorrectionEffect") then
        if obj.Brightness>0.4 then obj.Brightness=0 end
    elseif obj:IsA("BlurEffect") then obj.Size=0 end
end
local function hookFlashObj(obj)
    killFlashObj(obj)
    if obj:IsA("Frame") or obj:IsA("ImageLabel") then
        local c1=obj:GetPropertyChangedSignal("BackgroundTransparency"):Connect(function() if antiFlashEnabled then killFlashObj(obj) end end)
        local c2=obj:GetPropertyChangedSignal("BackgroundColor3"):Connect(function() if antiFlashEnabled then killFlashObj(obj) end end)
        table.insert(flashConns,c1); table.insert(flashConns,c2)
    elseif obj:IsA("ColorCorrectionEffect") then
        local c1=obj:GetPropertyChangedSignal("Brightness"):Connect(function() if antiFlashEnabled and obj.Brightness>0.4 then obj.Brightness=0 end end)
        table.insert(flashConns,c1)
    elseif obj:IsA("BlurEffect") then
        local c1=obj:GetPropertyChangedSignal("Size"):Connect(function() if antiFlashEnabled and obj.Size>0 then obj.Size=0 end end)
        table.insert(flashConns,c1)
    end
end
local function enableAntiFlash()
    for _,c in ipairs(flashConns) do pcall(function() c:Disconnect() end) end; flashConns={}
    local L=game:GetService("Lighting"); local pg=LocalPlayer:FindFirstChildOfClass("PlayerGui")
    if pg then
        for _,obj in ipairs(pg:GetDescendants()) do pcall(hookFlashObj,obj) end
        local c=pg.DescendantAdded:Connect(function(obj) if antiFlashEnabled then task.wait(); pcall(hookFlashObj,obj) end end)
        table.insert(flashConns,c)
    end
    for _,eff in ipairs(L:GetChildren()) do pcall(hookFlashObj,eff) end
    local lc=L.ChildAdded:Connect(function(obj) if antiFlashEnabled then task.wait(); pcall(hookFlashObj,obj) end end)
    table.insert(flashConns,lc)
end
local function disableAntiFlash()
    for _,c in ipairs(flashConns) do pcall(function() c:Disconnect() end) end; flashConns={}
    for _,eff in ipairs(game:GetService("Lighting"):GetChildren()) do if eff:IsA("BlurEffect") then eff.Size=24 end end
end

-- Store camera CFrame at start of every frame BEFORE game processes input
-- This lets no-recoil see exactly what changed between frames
RunService.RenderStepped:Connect(function()

    -- ── AIMBOT ────────────────────────────────────────────────────────────────
    -- Smooth lerp toward target while RMB held.
    -- Alpha = (smooth/10) clamped. smooth=1 → 0.1/frame (slow), smooth=10 → 1.0 (instant snap).
    if aimbotEnabled and RMBHeld then
        local _, worldPos = getBestTarget(aimbotFOV)
        if worldPos then
            local origin = Camera.CFrame.Position
            local _, onScreen = Camera:WorldToViewportPoint(worldPos)
            if onScreen then
                local alpha = math.clamp(aimbotSmooth / 10, 0.04, 1.0)
                local targetCF = CFrame.lookAt(origin, worldPos)
                Camera.CFrame = Camera.CFrame:Lerp(targetCF, alpha)
            end
        end
    end

    -- ── NO RECOIL ─────────────────────────────────────────────────────────────
    -- Must run AFTER aimbot (so we don't cancel aimbot's own camera movement)
    if noRecoilEnabled then
        pcall(applyNoRecoil)
    else
        noRecoilBaseCF = nil
    end

    -- ── FOV CIRCLES ───────────────────────────────────────────────────────────
    local vp = Camera.ViewportSize
    local ctr = Vector2.new(vp.X/2, vp.Y/2)
    fovCircle.Position = ctr; fovCircle.Radius = aimbotFOV
    fovCircle.Visible  = fovCircleVisible and aimbotEnabled
    silentFovCircle.Position = ctr; silentFovCircle.Radius = silentAimFOV
    silentFovCircle.Visible  = fovCircleVisible and silentAimEnabled

    -- ── CROSSHAIR ─────────────────────────────────────────────────────────────
    updateCrosshair()

    -- ── ESP ───────────────────────────────────────────────────────────────────
    for _,p in pairs(Players:GetPlayers()) do pcall(updatePlayerESP,p) end

    -- ── TRIGGERBOT ────────────────────────────────────────────────────────────
    pcall(doTriggerbot)

    -- ── STORE BASELINE FOR NEXT FRAME'S NO-RECOIL CHECK ─────────────────────
    -- We store at end of frame so we capture the camera AFTER aimbot + recoil cancel,
    -- giving a clean "intended" baseline for the next frame comparison.
    if noRecoilEnabled then
        noRecoilBaseCF = Camera.CFrame
    end
end)

RunService.Heartbeat:Connect(function()
    for _,p in pairs(Players:GetPlayers()) do

        end
    end
    if infiniteAmmoEnabled then pcall(patchAmmo) end
end)
